Animation

Simple example (FuncAnimation)

Working example that works for quarto

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from IPython.display import HTML

# Use the notebook backend for interactive plots
%matplotlib notebook

# Set up the figure and axes
fig, ax = plt.subplots()
xdata, ydata = [], []
ln, = plt.plot([], [], 'r-')

# Initialize the plot
def init():
    ax.set_xlim(0, 2 * np.pi)
    ax.set_ylim(-1, 1)
    return ln,

# Update function for animation
def update(frame):
    xdata.append(frame)
    ydata.append(np.sin(frame))
    ln.set_data(xdata, ydata)
    return ln,

# Create the animation
ani = FuncAnimation(
    fig, update, frames=np.linspace(0, 2 * np.pi, 128),
    init_func=init, blit=True, interval=50
)

# Display the animation
HTML(ani.to_html5_video())

3D Plot Animation (.py)

Run the following in a .py file.

import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import axes3d
from matplotlib import cm

# Enable interactive mode
plt.ion()

# Create a new figure
fig = plt.figure()
ax = fig.add_subplot(projection='3d')

# Generate test data
X, Y, Z = axes3d.get_test_data(0.05)

# Plot surface and contours
ax.plot_surface(X, Y, Z, rstride=8, cstride=8, alpha=0.3)
cset = ax.contour(X, Y, Z, zdir='z', offset=-100, cmap=cm.coolwarm)
cset = ax.contour(X, Y, Z, zdir='x', offset=-40, cmap=cm.coolwarm)
cset = ax.contour(X, Y, Z, zdir='y', offset=40, cmap=cm.coolwarm)

# Set labels and limits
ax.set_xlabel('X')
ax.set_xlim(-40, 40)
ax.set_ylabel('Y')
ax.set_ylim(-40, 40)
ax.set_zlabel('Z')
ax.set_zlim(-100, 100)

# Show the plot
plt.show()

# Keep the plot open
plt.ioff()
plt.show()
